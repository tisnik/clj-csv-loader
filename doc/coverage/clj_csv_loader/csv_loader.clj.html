013  (ns clj-csv-loader.csv-loader)
014  
015  (require '[clojure.data.csv :as csv])
016  (require '[clojure.java.io  :as io])
017  
018  (defn csv-data->maps
019    "Convert read CSV data into proper Clojure map."
020    [csv-data]
021    (map zipmap
022      (->> (first csv-data)  ;; header
023           (map keyword)     ;; heder items -> keywords
024           repeat)
025      (rest csv-data)))
026  
027  (defn load-csv
028    "Load CSV specified by filename and convert it to a proper Clojure map."
029    [filename]
030    (with-open [reader (io/reader filename)]
031      (let [data (csv/read-csv reader)]
032           (doall (csv-data->maps data)))))
033  
034  (defn load-csv-for-all-dates
035    "Load CSV files for all dates specified."
036    [dates-from data-directory filename]
037    (zipmap dates-from
038            (for [date dates-from]
039              (let [full-filename (str data-directory "/" date "/" filename)]
040                (load-csv full-filename)))))
